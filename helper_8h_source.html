588 {
589  double t2 = (y1 - y2 + tan(phi1) * (x2 - x1)) / (sin(phi2) - tan(phi1) * cos(phi2));
590  double t1 = (x2 - x1 + cos(phi2) * t2) / cos(phi1);
591 
592  if (t1 > 0)
593  type1 = 1;
594  if (t1 == 0)
595  type1 = 0;
596  if (t1 < 0)
597  type1 = -1;
598

602  type2 = 0;
603  if (t2 < 0)
604  type2 = -1;
605 
606  iPx = x1 + t1 * cos(phi1);
607  iPy = y1 + t1 * sin(phi1);
