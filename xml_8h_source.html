126 void appendLinkToNodeXercesC(nodeElement road, link &successor, link &predecessor)
127 {
128  if (successor.id == -1 && predecessor.id == -1){
129  return;
130  }
131  nodeElement link("link");
132  link.appendToNode(road);
133 
134  if( predecessor.id != -1){
135  nodeElement pre("predecessor");
136  pre.appendToNode(link);
137  pre.addAttribute("elementId", predecessor.id);
138  pre.addAttribute("elementType", getLinkType(predecessor.elementType).c_str());
139  if(predecessor.contactPoint != noneType)
140  pre.addAttribute("contactPoint", getContactPointType(predecessor.contactPoint).c_str());
141  }
142 
143  if( successor.id != -1){
144  nodeElement suc("successor");
145  suc.appendToNode(link);
146  suc.addAttribute("elementId", successor.id);
147  suc.addAttribute("elementType", getLinkType(successor.elementType));
148  if(successor.contactPoint != noneType)
149  suc.addAttribute("contactPoint", getContactPointType(successor.contactPoint));
150  }
151 }
